def read_from_db(DB_URI: str, query: str) -> list:
    """Hjelpefunksjon for å lese fra db"""

    # Connect to an existing database
    with psycopg.connect(DB_URI, row_factory=namedtuple_row) as conn:

        # Open a cursor to perform database operations
        with conn.cursor() as cur:

            # Execute a command
            cur.execute(query)
            data = cur.fetchall()

    return data

def create_read_query(sensors, start, stop):
    """
    Query for å hente timesaggregerte data for en sensor
    """
    
    query=f"""
    SELECT time_bucket AS "time", measurement, sensor_navn 
    FROM data.measurements_avg_1hour
    WHERE time_bucket BETWEEN '{start}' AND '{stop}'
    AND sensor_navn in ({sensors});
    """
    
    return query
